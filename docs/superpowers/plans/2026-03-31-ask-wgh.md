# Ask WGH Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a conversational AI dish finder that takes a natural language question and returns real dish recommendations from the database, powered by Claude Sonnet 4.6.

**Architecture:** Supabase Edge Function receives question + optional location/auth, pre-fetches ~300 ranked dishes via RPC, builds a context-stuffed prompt, calls Claude Sonnet 4.6 via raw HTTP, returns a natural language answer with parseable dish/restaurant links.

**Tech Stack:** Supabase Edge Functions (Deno), Anthropic Messages API (raw fetch), React, React Query (useMutation), Supabase RPC (rate limiting + data)

**Spec:** `docs/superpowers/specs/2026-03-31-ask-wgh-design.md`

```js
// Matches [[dish:ID|Name]] or [[restaurant:ID|Name]]
// ID must be UUID-like (alphanumeric + hyphens only) for security
const LINK_PATTERN = /\[\[(dish|restaurant):([a-f0-9-]+)\|([^\]]+)\]\]/g

export function parseAskResponse(text) {
  if (!text) return { segments: [], mentions: [] }

  const segments = []
  const mentions = []
  let lastIndex = 0

  // Reset regex state
  LINK_PATTERN.lastIndex = 0

  let match
  while ((match = LINK_PATTERN.exec(text)) !== null) {
    const [fullMatch, type, id, name] = match

    // Add preceding text if any
    if (match.index > lastIndex) {
      segments.push({ type: 'text', content: text.slice(lastIndex, match.index) })
    }

    segments.push({ type, id, name })
    mentions.push({ type, id, name })
    lastIndex = match.index + fullMatch.length
  }

  // Add remaining text
  if (lastIndex < text.length) {
    segments.push({ type: 'text', content: text.slice(lastIndex) })
  }

  return { segments, mentions }
}
```

```js
import { supabase } from '../lib/supabase'
import { createClassifiedError } from '../utils/errorHandler'
import { logger } from '../utils/logger'

export const askApi = {
  async askQuestion({ question, lat, lng }) {
    try {
      const session = await supabase.auth.getSession()
      const accessToken = session?.data?.session?.access_token

      const headers = {
        'Content-Type': 'application/json',
        apikey: import.meta.env.VITE_SUPABASE_ANON_KEY,
      }
      if (accessToken) {
        headers['Authorization'] = `Bearer ${accessToken}`
      }

      const response = await fetch(
        `${import.meta.env.VITE_SUPABASE_URL}/functions/v1/ask-wgh`,
        {
          method: 'POST',
          headers,
          body: JSON.stringify({ question, user_lat: lat, user_lng: lng }),
        }
      )

      if (!response.ok) {
        const errorData = await response.json().catch(() => ({}))
        if (response.status === 429) {
          const err = new Error(errorData.message || 'Too many asks. Try again later.')
          err.type = 'rate_limit'
          err.retryAfterSeconds = errorData.retry_after_seconds || 60
          err.remainingAsks = 0
          throw err
        }
        throw createClassifiedError(
          new Error(errorData.error || `Ask failed (${response.status})`)
        )
      }

      const data = await response.json()
      return data
    } catch (error) {
      logger.error('Ask WGH error:', error)
      throw error.type ? error : createClassifiedError(error)
    }
  },
}
```

Create `src/hooks/useAskWGH.js`:

```js
import { useState, useCallback } from 'react'
import { useMutation } from '@tanstack/react-query'
import { askApi } from '../api/askApi'
import { checkAskRateLimit } from '../lib/rateLimiter'
import { parseAskResponse } from '../utils/askParser'
import { useAuth } from '../context/AuthContext'
import { useLocation as useGeoLocation } from '../context/LocationContext'
import { logger } from '../utils/logger'

export function useAskWGH() {
  const { user } = useAuth()
  const { location } = useGeoLocation()
  const [parsedResponse, setParsedResponse] = useState(null)
  const [rateLimitInfo, setRateLimitInfo] = useState(null)

  const mutation = useMutation({
    mutationFn: (question) => {
      return askApi.askQuestion({
        question,
        lat: location?.lat || null,
        lng: location?.lng || null,
      })
    },
    onSuccess: (data) => {
      const parsed = parseAskResponse(data.answer)
      setParsedResponse(parsed)
      if (data.remaining_asks !== undefined) {
        setRateLimitInfo({ remaining: data.remaining_asks })
      }
    },
    onError: (error) => {
      if (error.type === 'rate_limit') {
        setRateLimitInfo({
          remaining: 0,
          retryAfterSeconds: error.retryAfterSeconds,
        })
      }
      logger.error('Ask WGH mutation error:', error)
    },
  })

  const ask = useCallback((question) => {
    // Client-side rate limit check first
    const rateCheck = checkAskRateLimit()
    if (!rateCheck.allowed) {
      setRateLimitInfo({
        remaining: 0,
        retryAfterSeconds: Math.ceil((rateCheck.retryAfterMs || 60000) / 1000),
      })
      return
    }

    setParsedResponse(null)
    mutation.mutate(question)
  }, [mutation])

  const reset = useCallback(() => {
    setParsedResponse(null)
    mutation.reset()
  }, [mutation])

  const maxAsks = user ? 6 : 2

  return {
    ask,
    reset,
    loading: mutation.isPending,
    error: mutation.error,
    parsedResponse,
    rateLimitInfo,
    isAuthenticated: !!user,
    maxAsks,
  }
}
```

```typescript
import { serve } from 'https://deno.land/std@0.177.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

const corsHeaders = {
  'Access-Control-Allow-Origin': 'https://wghapp.com',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

const ANTHROPIC_API_URL = 'https://api.anthropic.com/v1/messages'
const MODEL = 'claude-sonnet-4-6'
const MAX_TOKENS = 1024

// Guest rate limiting by IP (in-memory, resets on cold start)
const guestAsks = new Map<string, number[]>()
const GUEST_MAX = 2
const GUEST_WINDOW_MS = 3600000

function checkGuestRateLimit(ip: string): { allowed: boolean; retry_after_seconds?: number } {
  const now = Date.now()
  const timestamps = (guestAsks.get(ip) || []).filter((t) => now - t < GUEST_WINDOW_MS)
  if (timestamps.length >= GUEST_MAX) {
    const oldest = timestamps[0]
    const retryAfter = Math.ceil((oldest + GUEST_WINDOW_MS - now) / 1000)
    return { allowed: false, retry_after_seconds: retryAfter }
  }
  timestamps.push(now)
  guestAsks.set(ip, timestamps)
  return { allowed: true }
}

const SYSTEM_PROMPT = `You are the local dish expert for What's Good Here — a food discovery app.
You answer like a knowledgeable friend who's eaten everywhere and remembers what's great. Casual but articulate. Never stiff, never sloppy.

RULES:
- Only recommend dishes from the DATA below. Never invent dishes or restaurants.
- Default: one best pick for vague questions. If they want options, give top 3 with variety.
- No dead ends. If you can't match exactly, pivot: "No Ethiopian, but if you want bold flavors, try..."
- Communicate vote confidence honestly: "9.4 backed by 142 voters" vs "9.2 but only 4 votes — early word is strong."
- Use real prices when available. Fall back to vibe (casual vs fine dining) when not.
- For dietary questions, do your best from dish names/categories, always add "confirm with the restaurant."
- Never badmouth any restaurant or dish. Steer toward the good.
- Never give health or allergy advice beyond "check with the restaurant."
- Never share another user's data.
- Keep answers concise — 2-4 sentences for simple questions, up to a short paragraph for restaurant deep-dives.
- When mentioning a dish, format as: [[dish:DISH_UUID|Dish Name]]
- When mentioning a restaurant, format as: [[restaurant:RESTAURANT_UUID|Restaurant Name]]
- If the user asks about something outside food, briefly redirect: "I'm all about the food — but speaking of the harbor, have you tried..."
- NEVER use markdown formatting (no **, no ##, no bullet points). Write in plain conversational text.`

function formatDishData(dishes: any[]): string {
  if (!dishes || dishes.length === 0) return 'No dishes available.'
  return dishes
    .slice(0, 300)
    .map((d: any, i: number) => {
      const price = d.price ? `$${d.price}` : 'market'
      const rating = d.avg_rating ? d.avg_rating.toFixed(1) : '?'
      const votes = d.total_votes || 0
      const town = d.restaurant_town || d.town || ''
      const dist = d.distance_miles ? ` (${d.distance_miles.toFixed(1)}mi)` : ''
      return `#${i + 1} ${d.dish_name || d.name} | ${d.restaurant_name} | ${town}${dist} | ${price} | ${rating} (${votes} votes) | ${d.category || ''} | id:${d.dish_id || d.id} | rid:${d.restaurant_id}`
    })
    .join('\n')
}

function formatUserContext(votes: any[], stats: any): string {
  if (!votes || votes.length === 0) return ''

  const topVotes = votes
    .filter((v: any) => v.rating_10 >= 8)
    .slice(0, 5)
    .map((v: any) => `${v.dish_name} at ${v.restaurant_name} (${v.rating_10}/10)`)
    .join(', ')

  const totalVotes = votes.length
  const avgRating = votes.reduce((sum: number, v: any) => sum + (v.rating_10 || 0), 0) / totalVotes

  let context = `\nUSER CONTEXT:\n- Voted on ${totalVotes} dishes.`
  if (topVotes) context += ` Favorites: ${topVotes}.`
  context += `\n- Average rating: ${avgRating.toFixed(1)}/10.`

  if (stats?.preferred_categories) {
    context += `\n- Preferred categories: ${stats.preferred_categories.join(', ')}.`
  }

  return context
}

serve(async (req: Request) => {
  // CORS preflight
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders })
  }

  const startTime = Date.now()

  try {
    const { question, user_lat, user_lng } = await req.json()

    // Validate question
    if (!question || typeof question !== 'string' || question.trim().length === 0) {
      return new Response(
        JSON.stringify({ error: 'Question is required' }),
        { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      )
    }
    if (question.length > 500) {
      return new Response(
        JSON.stringify({ error: 'Question too long (max 500 characters)' }),
        { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      )
    }

    // Set up Supabase client
    const supabaseUrl = Deno.env.get('SUPABASE_URL')!
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
    const anthropicKey = Deno.env.get('ANTHROPIC_API_KEY')!

    const authHeader = req.headers.get('Authorization')
    let userId: string | null = null
    let remainingAsks: number | null = null

    // Auth + rate limiting
    if (authHeader) {
      const supabaseAuth = createClient(supabaseUrl, supabaseServiceKey, {
        global: { headers: { Authorization: authHeader } },
      })
      try {
        const { data: { user } } = await supabaseAuth.auth.getUser(
          authHeader.replace('Bearer ', '')
        )
        userId = user?.id || null
      } catch (_) {
        // Continue as guest
      }
    }

    const supabase = createClient(supabaseUrl, supabaseServiceKey)

    if (userId) {
      // Authenticated rate limit via RPC
      const { data: rateCheck } = await supabase.rpc('check_and_record_rate_limit', {
        p_action: 'ask_wgh',
        p_max_attempts: 6,
        p_window_seconds: 3600,
      })
      if (rateCheck && !rateCheck.allowed) {
        return new Response(
          JSON.stringify({
            error: 'Rate limit exceeded',
            message: rateCheck.message || 'Too many asks. Try again later.',
            retry_after_seconds: rateCheck.retry_after_seconds,
            remaining_asks: 0,
          }),
          { status: 429, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
        )
      }
      // Calculate remaining
      const { data: recentCount } = await supabase
        .from('rate_limits')
        .select('id', { count: 'exact', head: true })
        .eq('user_id', userId)
        .eq('action', 'ask_wgh')
        .gte('created_at', new Date(Date.now() - 3600000).toISOString())
      remainingAsks = Math.max(0, 6 - ((recentCount as any) || 0))
    } else {
      // Guest rate limit by IP
      const ip = req.headers.get('x-forwarded-for')?.split(',')[0]?.trim() || 'unknown'
      const guestCheck = checkGuestRateLimit(ip)
      if (!guestCheck.allowed) {
        return new Response(
          JSON.stringify({
            error: 'Rate limit exceeded',
            message: 'Guest limit reached. Sign in for 6 asks per hour.',
            retry_after_seconds: guestCheck.retry_after_seconds,
            remaining_asks: 0,
          }),
          { status: 429, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
        )
      }
    }

    // Fetch dish data
    const lat = user_lat || 41.3805  // MV center fallback
    const lng = user_lng || -70.6456
    const { data: dishes, error: dishError } = await supabase.rpc('get_ranked_dishes', {
      user_lat: lat,
      user_lng: lng,
      radius_miles: 50,
      filter_category: null,
      filter_town: null,
    })
    if (dishError) {
      console.error('Dish fetch error:', dishError)
    }

    // Fetch user context if authenticated
    let userContextStr = ''
    if (userId) {
      const { data: votes } = await supabase
        .from('votes')
        .select('rating_10, dish_id, dishes(name, restaurants(name))')
        .eq('user_id', userId)
        .eq('source', 'user')
        .order('created_at', { ascending: false })
        .limit(50)

      const formattedVotes = (votes || []).map((v: any) => ({
        rating_10: v.rating_10,
        dish_name: v.dishes?.name,
        restaurant_name: v.dishes?.restaurants?.name,
      }))

      const { data: stats } = await supabase
        .from('profiles')
        .select('preferred_categories')
        .eq('id', userId)
        .maybeSingle()

      userContextStr = formatUserContext(formattedVotes, stats)
    }

    // Build prompt
    const dishDataStr = formatDishData(dishes || [])
    const userMessage = `DATA:\n${dishDataStr}${userContextStr}\n\nQUESTION: ${question.trim()}`

    // Call Claude
    const claudeResponse = await fetch(ANTHROPIC_API_URL, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': anthropicKey,
        'anthropic-version': '2023-06-01',
      },
      body: JSON.stringify({
        model: MODEL,
        max_tokens: MAX_TOKENS,
        system: SYSTEM_PROMPT,
        messages: [{ role: 'user', content: userMessage }],
      }),
    })

    if (!claudeResponse.ok) {
      const errBody = await claudeResponse.text()
      console.error('Claude API error:', claudeResponse.status, errBody)
      return new Response(
        JSON.stringify({ error: 'AI service temporarily unavailable' }),
        { status: 502, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      )
    }

    const claudeData = await claudeResponse.json()
    const answer = claudeData.content?.[0]?.text || 'Sorry, I couldn\'t come up with a recommendation right now.'

    const responseTimeMs = Date.now() - startTime

    // Log for cost tracking (fire and forget)
    const inputTokens = claudeData.usage?.input_tokens || 0
    const outputTokens = claudeData.usage?.output_tokens || 0

    supabase.from('ask_logs').insert({
      user_id: userId,
      question_hash: await hashQuestion(question),
      tokens_input: inputTokens,
      tokens_output: outputTokens,
      response_time_ms: responseTimeMs,
      dishes_mentioned: (answer.match(/\[\[dish:/g) || []).length,
    }).then(() => {}).catch((e: any) => console.error('Log insert failed:', e))

    return new Response(
      JSON.stringify({
        answer,
        remaining_asks: remainingAsks,
      }),
      { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    )
  } catch (error) {
    console.error('Ask WGH error:', error)
    return new Response(
      JSON.stringify({ error: 'Internal server error' }),
      { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    )
  }
})

async function hashQuestion(question: string): Promise<string> {
  const encoder = new TextEncoder()
  const data = encoder.encode(question.toLowerCase().trim())
  const hashBuffer = await crypto.subtle.digest('SHA-256', data)
  const hashArray = Array.from(new Uint8Array(hashBuffer))
  return hashArray.map((b) => b.toString(16).padStart(2, '0')).join('')
}
```
